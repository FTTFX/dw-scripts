local lines, order = {}, 0
local function addLine(txt, col)
    order = order + 1
    local l = Instance.new("TextLabel")
    l.Size = UDim2.new(1, -12, 0, 16) l.BackgroundTransparency = 1
    l.Font = Enum.Font.Code l.TextSize = 12 l.TextXAlignment = Enum.TextXAlignment.Left
    l.TextColor3 = col or Color3.fromRGB(210,210,210) l.Text = txt l.LayoutOrder = order l.Parent = scroll
    table.insert(lines, txt)
    scroll.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + 10)
end

copyBtn.MouseButton1Click:Connect(function()
    local full = table.concat(lines, "\n")
    local clip = setclipboard or toclipboard or (getgenv and getgenv().setclipboard)
    local ok = clip and pcall(clip, full)
    if ok then copyBtn.Text = "Copied!" task.delay(2, function() if copyBtn.Parent then copyBtn.Text="Copy" end end)
    else copyBox.Visible = true copyBox.Text = full copyBox:CaptureFocus() copyBtn.Text = "Ctrl+C" end
end)

local function scan()
    for _, c in pairs(scroll:GetChildren()) do if c:IsA("TextLabel") then c:Destroy() end end
    lines = {} order = 0
    copyBox.Visible = false
    addLine("=== INVENTORY SPY ===", Color3.fromRGB(120,220,255))
    local igp = workspace:FindFirstChild("InGamePlayers")
    local entry = igp and igp:FindFirstChild(LP.Name)
    local inv = entry and entry:FindFirstChild("Inventory")
    if not inv then
        addLine("❌ ไม่เจอ Inventory (igp=" .. tostring(igp ~= nil) .. " entry=" .. tostring(entry ~= nil) .. ")", Color3.fromRGB(255,90,90))
        return
    end
    addLine("path: " .. inv:GetFullName(), Color3.fromRGB(140,140,140))
    addLine("", Color3.fromRGB(140,140,140))
    local n = 0
    for _, s in pairs(inv:GetChildren()) do
        n = n + 1
        local val = "?"
        if s:IsA("ValueBase") then
            if s:IsA("ObjectValue") then val = s.Value and s.Value:GetFullName() or "nil"
            else val = tostring(s.Value) end
        end
        addLine(string.format("[%s]  %s  =  '%s'", s.ClassName, s.Name, val),
            (s.Name:match("^Slot%d+$")) and Color3.fromRGB(120,255,140) or Color3.fromRGB(200,200,200))
    end
    if n == 0 then addLine("(Inventory ว่างเปล่า ไม่มี children)", Color3.fromRGB(255,160,50)) end
    addLine("", Color3.fromRGB(140,140,140))
    addLine("=== รวม " .. n .. " children ===", Color3.fromRGB(120,220,255))
end

rescanBtn.MouseButton1Click:Connect(scan)
scan()
warn("[DW_INV] Inventory Spy v1.0 — ดูค่าช่องกระเป๋า (สีเขียว=Slot จริง) กด Copy ส่งได้")

-- 03Forsaken_saehook.txt — Forsaken sae script reverse-hook v1.0
-- รันตัวนี้ "ก่อน" sae script → hook ทุก action → ดูว่า sae ปั่น Gen ยังไง
--
-- วิธีใช้:
--   1. โหลด hook นี้
--   2. โหลด sae: loadstring(game:HttpGet("https://raw.githubusercontent.com/MaybeNotMandy/forsaken/refs/heads/main/sae"))()
--   3. กด Auto-Gen ใน sae GUI 1-2 นาที (ลองยืนใกล้/ไกล gen)
--   4. กด Copy → ส่งให้ผม
--
-- Hook:
--   1. __namecall: FireServer / InvokeServer / fireproximityprompt
--   2. ProximityPrompt: Triggered / TriggerEnded / InputHoldBegin / HoldDuration get/set
--   3. RemoteEvent:FireServer + Reliable / UnreliableRemoteEvent (ByteNet detect)
--   4. RunService.BindToRenderStep (ดู priority สูงแค่ไหน)
--   5. Humanoid.WalkSpeed write
--   6. workspace.Map.<...>.Generator descendant access (ทาง index path)

-- ────────── 2) Log buffer ──────────
local LOGS = { remote = {}, prompt = {}, render = {}, walk = {}, all = {} }
local MAX_PER_CAT = 400
local startClock = os.clock()
local function ts() return string.format("%.3f", os.clock() - startClock) end

local function push(cat, line)
    local t = LOGS[cat]
    t[#t+1] = "[" .. ts() .. "] " .. line
    if #t > MAX_PER_CAT then table.remove(t, 1) end
    -- also append to all
    if cat ~= "all" then
        LOGS.all[#LOGS.all+1] = "[" .. cat .. "][" .. ts() .. "] " .. line
        if #LOGS.all > MAX_PER_CAT * 4 then table.remove(LOGS.all, 1) end
    end
end

-- ────────── 3) Value-to-string (deep) ──────────
local function shortPath(inst, maxDepth)
    if not inst then return "(nil)" end
    local ok, full = pcall(function() return inst:GetFullName() end)
    if not ok then return tostring(inst) end
    -- shorten if too long
    local parts = {}
    for p in full:gmatch("[^%.]+") do parts[#parts+1] = p end
    if #parts > (maxDepth or 5) then
        return "..." .. table.concat(parts, ".", #parts - (maxDepth or 5) + 1)
    end
    return full
end

local function v2s(v, depth)
    depth = depth or 0
    if depth > 3 then return "..." end
    local t = typeof(v)
    if t == "string" then
        if #v > 80 then return string.format("%q..(%d)", v:sub(1,77), #v) end
        return string.format("%q", v)
    end
    if t == "Instance" then return "<" .. v.ClassName .. ":" .. shortPath(v, 4) .. ">" end
    if t == "buffer" then
        local ok, len = pcall(function() return buffer.len(v) end)
        local hex = ""
        if ok and len then
            local n = math.min(len, 32)
            local bs = {}
            for i = 0, n-1 do bs[#bs+1] = string.format("%02x", buffer.readu8(v, i)) end
            hex = table.concat(bs, "")
            return string.format("<buffer:%dB %s%s>", len, hex, (len > n and ".." or ""))
        end
        return "<buffer>"
    end
    if t == "Vector3" then return string.format("V3(%.1f,%.1f,%.1f)", v.X, v.Y, v.Z) end
    if t == "CFrame"  then return "CF(" .. tostring(v.Position) .. ")" end
    if t == "table" then
        local parts = {}
        local n = 0
        for k, val in pairs(v) do
            n = n + 1
            if n > 8 then parts[#parts+1] = "..."; break end
            parts[#parts+1] = tostring(k) .. "=" .. v2s(val, depth + 1)
        end
        return "{" .. table.concat(parts, ", ") .. "}"
    end
    return tostring(v)
end

local function dumpArgs(...)
    local n = select("#", ...)
    if n == 0 then return "" end
    local out = {}
    for i = 1, math.min(n, 6) do
        out[i] = v2s((select(i, ...)))
    end
    if n > 6 then out[#out+1] = "...(" .. (n-6) .. " more)" end
    return table.concat(out, ", ")
end

-- ────────── 4) Hooks ──────────
local restores = {}

-- (a) __namecall — catch FireServer/InvokeServer/fireproximityprompt
local mt = getrawmetatable(game)
local origNamecall = mt.__namecall
local oldRO = nil
pcall(function() oldRO = setreadonly and setreadonly(mt, false) end)

mt.__namecall = newcclosure(function(self, ...)
    local method = getnamecallmethod()
    if method == "FireServer" or method == "InvokeServer" then
        local cls = "?"
        pcall(function() cls = self.ClassName end)
        push("remote", string.format("%s:%s(%s)  via %s",
            cls, method, dumpArgs(...), shortPath(self, 4)))
    elseif method == "fireproximityprompt"
        or method == "InputHoldBegin" or method == "InputHoldEnd" then
        push("prompt", string.format("%s:%s(%s)  via %s",
            self.ClassName, method, dumpArgs(...), shortPath(self, 4)))
    end
    return origNamecall(self, ...)
end)
restores[#restores+1] = function()
    pcall(function() mt.__namecall = origNamecall end)
end
if setreadonly and oldRO ~= nil then pcall(function() setreadonly(mt, true) end) end

-- (b) global fireproximityprompt (executor function — not always going via :)
if rawget(getfenv(), "fireproximityprompt") then
    local origFire = fireproximityprompt
    getgenv().fireproximityprompt = function(prompt, count)
        push("prompt", string.format("fireproximityprompt(%s, %s)",
            shortPath(prompt, 4), tostring(count)))
        return origFire(prompt, count)
    end
    restores[#restores+1] = function()
        getgenv().fireproximityprompt = origFire
    end
end

-- (c) hookfunction on BindToRenderStep — ดู priority
if hookfunction then
    local origBind = RS.BindToRenderStep
    RS.BindToRenderStep = hookfunction(RS.BindToRenderStep, function(self, name, prio, fn)
        push("render", string.format("BindToRenderStep('%s', %s)", tostring(name), tostring(prio)))
        return origBind(self, name, prio, fn)
    end)
    restores[#restores+1] = function()
        pcall(function() hookfunction(RS.BindToRenderStep, origBind) end)
    end
end

-- (d) ProximityPrompt:Triggered (server confirmed gen tick)
local function bindPromptSignals(prompt)
    local function safeCon(sig, evt)
        pcall(function()
            sig:Connect(function(player)
                push("prompt", string.format("%s ← %s @ %s",
                    evt, player and player.Name or "?", shortPath(prompt, 4)))
            end)
        end)
    end
    safeCon(prompt.Triggered, "Triggered")
    safeCon(prompt.TriggerEnded, "TriggerEnded")
    pcall(function()
        prompt.PromptShown:Connect(function()
            push("prompt", "PromptShown @ " .. shortPath(prompt, 4))
        end)
    end)
end

-- existing prompts
for _, d in pairs(workspace:GetDescendants()) do
    if d:IsA("ProximityPrompt") then bindPromptSignals(d) end
end
-- future prompts
local descCon = workspace.DescendantAdded:Connect(function(d)
    if d:IsA("ProximityPrompt") then
        task.wait(0.1)
        bindPromptSignals(d)
    end
end)
restores[#restores+1] = function() pcall(function() descCon:Disconnect() end) end

-- (e) WalkSpeed write → ดูค่าที่ sae set
local function hookWalkSpeed(hum)
    pcall(function()
        hum:GetPropertyChangedSignal("WalkSpeed"):Connect(function()
            push("walk", string.format("WalkSpeed=%s on %s",
                tostring(hum.WalkSpeed), shortPath(hum, 4)))
        end)
        hum:GetPropertyChangedSignal("JumpPower"):Connect(function()
            push("walk", string.format("JumpPower=%s on %s",
                tostring(hum.JumpPower), shortPath(hum, 4)))
        end)
    end)
end
if LP.Character then
    local h = LP.Character:FindFirstChildOfClass("Humanoid")
    if h then hookWalkSpeed(h) end
end
restores[#restores+1] = (LP.CharacterAdded:Connect(function(c)
    task.wait(0.3)
    local h = c:FindFirstChildOfClass("Humanoid")
    if h then hookWalkSpeed(h) end
end))
    and function() end  -- placeholder

local function refreshTabBtns()
    for n, b in pairs(tabBtns) do
        b.BackgroundColor3 = (n == activeTab)
            and Color3.fromRGB(80, 130, 200) or Color3.fromRGB(30, 50, 80)
    end
end
refreshTabBtns()

local function renderTab()
    scroll:ClearAllChildren()
    layout = Instance.new("UIListLayout"); layout.Padding = UDim.new(0, 1); layout.Parent = scroll
    pad = Instance.new("UIPadding"); pad.PaddingTop = UDim.new(0, 4); pad.PaddingLeft = UDim.new(0, 6); pad.Parent = scroll
    if activeTab == "COPY" then
        -- show big TextBox with everything for copy
        local box = Instance.new("TextBox")
        box.Size = UDim2.new(1, -8, 1, -8); box.Position = UDim2.new(0, 4, 0, 4)
        box.MultiLine = true; box.TextWrapped = true; box.ClearTextOnFocus = false
        box.BackgroundColor3 = Color3.fromRGB(5, 8, 14)
        box.TextColor3 = Color3.fromRGB(220, 230, 240)
        box.TextSize = 11; box.Font = Enum.Font.Code
        box.TextXAlignment = Enum.TextXAlignment.Left
        box.TextYAlignment = Enum.TextYAlignment.Top
        local s = {}
        s[#s+1] = "=== FORSAKEN sae HOOK DUMP ==="
        s[#s+1] = string.format("remote=%d  prompt=%d  render=%d  walk=%d",
            #LOGS.remote, #LOGS.prompt, #LOGS.render, #LOGS.walk)
        s[#s+1] = ""
        s[#s+1] = "=== REMOTE ==="
        for _, l in ipairs(LOGS.remote) do s[#s+1] = l end
        s[#s+1] = ""
        s[#s+1] = "=== PROMPT ==="
        for _, l in ipairs(LOGS.prompt) do s[#s+1] = l end
        s[#s+1] = ""
        s[#s+1] = "=== RENDER ==="
        for _, l in ipairs(LOGS.render) do s[#s+1] = l end
        s[#s+1] = ""
        s[#s+1] = "=== WALK ==="
        for _, l in ipairs(LOGS.walk) do s[#s+1] = l end
        box.Text = table.concat(s, "\n")
        box.Parent = scroll
        pcall(function() if setclipboard then setclipboard(box.Text) end end)
        return
    end
    local source = LOGS[activeTab:lower()] or LOGS.all
    for _, line in ipairs(source) do
        local lbl = Instance.new("TextLabel")
        lbl.Size = UDim2.new(1, -12, 0, 14)
        lbl.BackgroundTransparency = 1
        lbl.TextColor3 = Color3.fromRGB(220, 230, 240)
        lbl.Text = line; lbl.TextScaled = false; lbl.TextSize = 11
        lbl.Font = Enum.Font.Code
        lbl.TextXAlignment = Enum.TextXAlignment.Left
        lbl.TextWrapped = true
        lbl.Parent = scroll
    end
end

for name, b in pairs(tabBtns) do
    b.MouseButton1Click:Connect(function()
        activeTab = name; refreshTabBtns(); renderTab()
    end)
end

-- Refresh stats + render
task.spawn(function()
    while gui.Parent do
        stats.Text = string.format("remote=%d  prompt=%d  render=%d  walk=%d  | t=%ss",
            #LOGS.remote, #LOGS.prompt, #LOGS.render, #LOGS.walk, ts())
        if activeTab ~= "COPY" then renderTab() end
        task.wait(1.5)
    end
end)

-- ────────── 6) Restore on close ──────────

    for _, r in ipairs(restores) do pcall(r) end
end
gui.AncestryChanged:Connect(function()
    if not gui.Parent then
        _G.FRSN_SAEHOOK_RESTORE()
    end
end)

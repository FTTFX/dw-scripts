-- ────── Logger ──────
local LOG = {}
local C = { remote = 0, prompt = 0, walk = 0 }
local startClock = os.clock()
local function ts() return string.format("%.1f", os.clock() - startClock) end
local function add(cat, line)
    C[cat] = (C[cat] or 0) + 1
    local s = "[" .. cat .. "][" .. ts() .. "] " .. tostring(line)
    LOG[#LOG+1] = s
    if #LOG > 1500 then table.remove(LOG, 1) end
    pcall(print, s)
    log.Text = s:sub(1, 38)
end

-- ────── HOOK ทันที (ใน pcall ทุกชั้น) ──────
local stageOK = { p = false, fpp = false, nc = false }

-- (a) ProximityPrompt.Triggered
do
    local n = 0
    pcall(function()
        for _, d in pairs(workspace:GetDescendants()) do
            if d:IsA("ProximityPrompt") then
                n = n + 1
                pcall(function()
                    d.Triggered:Connect(function(plr)
                        add("prompt", "Trig " .. (plr and plr.Name or "?") .. " " .. tostring(d.Parent and d.Parent.Name or "?"))
                    end)
                end)
            end
        end
    end)
    stageOK.p = (n > 0)
    add("walk", "found " .. n .. " ProximityPrompts")
end

-- (b) global fireproximityprompt
pcall(function()
    local g = getgenv and getgenv() or _G
    if rawget(g, "fireproximityprompt") then
        local orig = g.fireproximityprompt
        g.fireproximityprompt = function(p, c)
            pcall(function() add("prompt", "FPP " .. tostring(p and p:GetFullName() or "?") .. " ct=" .. tostring(c)) end)
            return orig(p, c)
        end
        stageOK.fpp = true
    end
end)

-- (c) __namecall
local ncErr = ""
pcall(function()
    local mt = getrawmetatable(game)
    if not mt then error("no mt") end
    local origNC = mt.__namecall
    if not origNC then error("no NC") end
    if setreadonly then pcall(setreadonly, mt, false) end
    mt.__namecall = function(self, ...)
        local method
        pcall(function() method = getnamecallmethod() end)
        if method == "FireServer" then
            pcall(function()
                local cls = self.ClassName
                local path = self:GetFullName()
                local n = select("#", ...)
                add("remote", cls .. ":FS args=" .. n .. " " .. path)
            end)
        elseif method == "InvokeServer" then
            pcall(function() add("remote", "RF:IS " .. self:GetFullName()) end)
        end
        return origNC(self, ...)
    end
    if setreadonly then pcall(setreadonly, mt, true) end
    stageOK.nc = true
end, function(e) ncErr = tostring(e) end)
if not stageOK.nc then add("walk", "NC FAIL: " .. ncErr) end

-- (d) WalkSpeed poll
local lastWS
task.spawn(function()
    while sg.Parent do
        task.wait(1)
        pcall(function()
            local ch = LP.Character
            if ch then
                local h = ch:FindFirstChildOfClass("Humanoid")
                if h and h.WalkSpeed ~= lastWS then
                    add("walk", "WalkSpeed=" .. h.WalkSpeed)
                    lastWS = h.WalkSpeed
                end
            end
        end)
    end
end)

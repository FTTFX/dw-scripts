-- ── Lighting backup (เก็บค่าเดิมก่อนแก้) ──────────────────────────
local LIG_BACKUP = nil
local function applyBright(on)
    if on then
        if not LIG_BACKUP then
            LIG_BACKUP = {
                FogEnd     = LIG.FogEnd,
                FogStart   = LIG.FogStart,
                Brightness = LIG.Brightness,
                Ambient    = LIG.Ambient,
                OutdoorAmbient = LIG.OutdoorAmbient,
                ClockTime  = LIG.ClockTime,
                atmosphere = {},
            }
            for _, a in pairs(LIG:GetChildren()) do
                if a:IsA("Atmosphere") then
                    LIG_BACKUP.atmosphere[a] = {
                        Density = a.Density, Haze = a.Haze,
                        Glare = a.Glare, Offset = a.Offset,
                    }
                end
            end
        end
        pcall(function()
            LIG.FogEnd     = 100000
            LIG.FogStart   = 100000
            LIG.Brightness = 3
            LIG.Ambient    = Color3.fromRGB(200, 200, 200)
            LIG.OutdoorAmbient = Color3.fromRGB(200, 200, 200)
            LIG.ClockTime  = 14
            for _, a in pairs(LIG:GetChildren()) do
                if a:IsA("Atmosphere") then
                    a.Density = 0
                    a.Haze    = 0
                    a.Glare   = 0
                end
            end
        end)
    else
        if LIG_BACKUP then
            pcall(function()
                LIG.FogEnd     = LIG_BACKUP.FogEnd
                LIG.FogStart   = LIG_BACKUP.FogStart
                LIG.Brightness = LIG_BACKUP.Brightness
                LIG.Ambient    = LIG_BACKUP.Ambient
                LIG.OutdoorAmbient = LIG_BACKUP.OutdoorAmbient
                LIG.ClockTime  = LIG_BACKUP.ClockTime
                for a, vals in pairs(LIG_BACKUP.atmosphere) do
                    if a and a.Parent then
                        a.Density = vals.Density
                        a.Haze    = vals.Haze
                        a.Glare   = vals.Glare
                        a.Offset  = vals.Offset
                    end
                end
            end)
            LIG_BACKUP = nil
        end
    end
end

-- ── Cleanup function (ใช้ตอนปิด) ─────────────────────────────────
local function cleanup()
    -- disconnect connections
    for _, c in pairs(_G.DW05_CONNS) do pcall(function() c:Disconnect() end) end
    _G.DW05_CONNS = {}
    -- ลบ Highlight
    for _, h in pairs(_G.DW05_HLS) do pcall(function() h:Destroy() end) end
    _G.DW05_HLS = {}
    -- คืน speed + noclip
    local char = LP.Character
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then
            hum.WalkSpeed = 16
            hum.PlatformStand = false
        end
        for _, p in pairs(char:GetDescendants()) do
            if p:IsA("BasePart") then p.CanCollide = true end
        end
    end
    -- ลบ BodyVelocity
    pcall(function() if _G.DW05_BV then _G.DW05_BV:Destroy() _G.DW05_BV = nil end end)
    -- คืน Lighting
    applyBright(false)
    -- ลบ GUI
    pcall(function() if _G.DW05_GUI then _G.DW05_GUI:Destroy() _G.DW05_GUI = nil end end)
    warn("[DW05] ปิดแล้ว")
end

-- ── หา paths (dynamic — รองรับทุก map name) ──────────────────────
local function getFolder(name)
    -- หาใน CurrentRoom ก่อน (ไม่สนชื่อ map)
    local currentRoom = workspace:FindFirstChild("CurrentRoom")
    if currentRoom then
        for _, map in pairs(currentRoom:GetChildren()) do
            local f = map:FindFirstChild(name)
            if f then return f end
        end
    end
    -- fallback: scan workspace ทั้งหมด
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj:IsA("Folder") and obj.Name == name then return obj end
    end
    return nil
end

-- ── ESP: สร้าง Highlight ──────────────────────────────────────────
local highlights = {}  -- [model] = Highlight

local function makeHL(model, fillColor, outlineColor, labelText)
    if highlights[model] then return end

    local container = {}  -- เก็บทุก instance ที่สร้าง

    -- Highlight (ทะลุกำแพง)
    pcall(function()
        local h = Instance.new("Highlight")
        h.Adornee             = model
        h.FillColor           = fillColor
        h.OutlineColor        = outlineColor
        h.FillTransparency    = 1    -- v1.3: ไม่มี fill เรืองแสง (กันแสบตา) — เส้นขอบอย่างเดียว
        h.OutlineTransparency = 0
        h.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
        h.Parent              = model
        table.insert(container, h)
        table.insert(_G.DW05_HLS, h)
    end)

    -- v1.3: ตัด PointLight ออก (แสงสว่างจ้า = แสบตา) — ใช้แค่เส้นขอบ + ป้ายชื่อ

    -- BillboardGui (ป้ายชื่อ)
    pcall(function()
        local root = model:FindFirstChild("HumanoidRootPart")
            or model:FindFirstChildWhichIsA("BasePart")
        if root then
            local bb = Instance.new("BillboardGui")
            bb.Size            = UDim2.new(0, 100, 0, 28)
            bb.StudsOffset     = Vector3.new(0, 3, 0)
            bb.AlwaysOnTop     = true  -- เห็นทะลุกำแพง
            bb.Parent          = root

            local lbl = Instance.new("TextLabel", bb)
            lbl.Size            = UDim2.new(1, 0, 1, 0)
            lbl.BackgroundTransparency = 1
            lbl.TextColor3      = outlineColor
            lbl.TextStrokeColor3 = Color3.new(0,0,0)
            lbl.TextStrokeTransparency = 0.4
            lbl.Text            = labelText or model.Name
            lbl.Font            = Enum.Font.GothamBold
            lbl.TextSize        = 14

            table.insert(container, bb)
            table.insert(_G.DW05_HLS, bb)
        end
    end)

    highlights[model] = container
end

local function removeHL(model)
    if highlights[model] then
        for _, inst in pairs(highlights[model]) do
            pcall(function() inst:Destroy() end)
        end
        highlights[model] = nil
    end
end

local lastESPTick = 0
local function updateESP()
    if not CFG.ESP_ON then return end
    if tick() - lastESPTick < 1 then return end  -- refresh ทุก 1s
    lastESPTick = tick()

    -- Generator ESP (เขียว)
    local genFolder = getFolder("Generators")
    if genFolder then
        for _, gen in pairs(genFolder:GetChildren()) do
            local completed = gen:FindFirstChild("Completed", true)
            local isDone = completed and completed:IsA("BoolValue") and completed.Value
            if not isDone then
                local cur = gen:FindFirstChild("CurrentAmount", true)
                local req = gen:FindFirstChild("RequiredAmount", true)
                local prog = (cur and req) and (cur.Value.."/"..req.Value) or ""
                makeHL(gen, Color3.fromRGB(0, 255, 80), Color3.fromRGB(0, 255, 80), "⚙ Gen "..prog)
            else
                removeHL(gen)
            end
        end
    end

    -- Twisted ESP (แดง)
    local monFolder = getFolder("Monsters")
    if monFolder then
        for _, mon in pairs(monFolder:GetChildren()) do
            local hum = mon:FindFirstChildOfClass("Humanoid")
            if hum and hum.Health > 0 then
                makeHL(mon, Color3.fromRGB(255, 50, 50), Color3.fromRGB(255, 120, 120), "⚠ "..mon.Name)
            else
                removeHL(mon)
            end
        end
    end

    -- Player ESP (เหลือง — เพื่อนคนอื่น)
    for _, pl in pairs(game:GetService("Players"):GetPlayers()) do
        if pl ~= LP then
            local ch = pl.Character
            local hum = ch and ch:FindFirstChildOfClass("Humanoid")
            if ch and hum and hum.Health > 0 then
                makeHL(ch, Color3.fromRGB(255, 230, 50), Color3.fromRGB(255, 230, 100), "👤 "..pl.Name)
            elseif ch then
                removeHL(ch)
            end
        end
    end

    -- ลบ highlight ของ model ที่หายไปแล้ว
    for model, container in pairs(highlights) do
        if not model.Parent then
            for _, inst in pairs(container) do
                pcall(function() inst:Destroy() end)
            end
            highlights[model] = nil
        end
    end
end

-- ── Noclip: ใช้ Stepped (runs ก่อน physics = ผลทันที) ─────────────
local noclipConn = RS.Stepped:Connect(function()
    if not CFG.NOCLIP_ON then return end
    local char = LP.Character
    if not char then return end
    for _, p in pairs(char:GetDescendants()) do
        if p:IsA("BasePart") then p.CanCollide = false end
    end
end)
table.insert(_G.DW05_CONNS, noclipConn)

-- ── Speed + ESP loop (Heartbeat) ─────────────────────────────────
local conn = RS.Heartbeat:Connect(function()
    local char = LP.Character
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then return end

    -- Speed
    hum.WalkSpeed = CFG.SPEED_ON and CFG.SPEED_VAL or 16

    -- ESP update
    updateESP()
end)
table.insert(_G.DW05_CONNS, conn)

-- ── ESP cleanup เมื่อขึ้นชั้นใหม่ ────────────────────────────────
table.insert(_G.DW05_CONNS,
    LP.CharacterAdded:Connect(function()
        for model, hl in pairs(highlights) do
            pcall(function() hl:Destroy() end)
        end
        highlights = {}
        _G.DW05_HLS = {}
        lastESPTick = 0
    end)
)

-- helper สร้างปุ่ม toggle
local function makeToggle(parent, label, yPos, getVal, setVal, onColor, offColor)
    local btn = Instance.new("TextButton", parent)
    btn.Size = UDim2.new(0, 90, 0, 20)
    btn.Position = UDim2.new(0, 2, 0, yPos)
    btn.Font = Enum.Font.Code
    btn.TextSize = 11
    btn.BorderSizePixel = 0
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 4)

    local function refresh()
        local v = getVal()
        btn.Text = label .. ": " .. (v and "ON" or "OFF")
        btn.BackgroundColor3 = v and onColor or Color3.fromRGB(60,60,60)
        btn.TextColor3 = v and Color3.new(0,0,0) or Color3.fromRGB(180,180,180)
    end
    btn.MouseButton1Click:Connect(function()
        setVal(not getVal())
        refresh()
    end)
    refresh()
    return btn, refresh
end

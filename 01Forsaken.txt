-- ==================== KICK-GUARD (บล็อก client-side anti-cheat kick) ====================
-- โดนเตะตอน Noclip/Climb = client AC (Systems.Character) เรียก LP:Kick()
-- → ดัก __namecall เมธอด "Kick" บนตัวเรา → บล็อก + log ว่าใครเรียก (เห็นต้นตอใน console)
-- hook ครั้งเดียว (flag กัน re-run stack) | executor ไม่มี hookmetamethod = ข้ามเงียบ
if not _G.FRSN_KICKGUARD then
    _G.FRSN_KICKGUARD = true
    local installed = pcall(function()
        local hookmeta = rawget(getfenv(), "hookmetamethod") or hookmetamethod
        local getnc    = rawget(getfenv(), "getnamecallmethod") or getnamecallmethod
        if not (hookmeta and getnc) then error("no hookmetamethod") end
        local oldNC
        oldNC = hookmeta(game, "__namecall", function(self, ...)
            local m = getnc()
            if (m == "Kick" or m == "kick") and self == _G.FRSN_LP then
                local tb = ""
                pcall(function() tb = debug.traceback() end)
                warn("[FRSN] 🛡️ BLOCKED Kick! ต้นตอ:\n" .. tb)
                return nil   -- กลืน → ไม่เตะ
            end
            return oldNC(self, ...)
        end)
        print("[FRSN] kick-guard ทำงาน (hookmetamethod __namecall)")
    end)
    if not installed then
        warn("[FRSN] kick-guard ติดตั้งไม่ได้ — executor ไม่ support hookmetamethod")
    end
end
_G.FRSN_LP = LP   -- kick-guard closure อ้าง _G.FRSN_LP (อัปเดตได้แม้ re-run โดยไม่ re-hook)

-- ==================== Role detect (Forsaken pattern) ====================
-- v2.4: ใช้ workspace.Players.<folder> เป็น signal หลัก (spy confirmed)
--   workspace.Players/
--   ├── Survivors/  (Model ชื่อ = char.Name ของ survivor)
--   ├── Killers/    (Model ชื่อ = char.Name ของ killer)
--   └── Spectating/ (Model ชื่อ = player.Name ของคนนั่งดู)
local function detectRole(player, char)
    if not char then return "survivor" end
    local wsP = workspace:FindFirstChild("Players")
    if wsP then
        -- 1) เช็คโดยตรง: char เป็น descendant ของ folder?
        local killers = wsP:FindFirstChild("Killers")
        if killers and char:IsDescendantOf(killers) then return "killer" end
        local survivors = wsP:FindFirstChild("Survivors")
        if survivors and char:IsDescendantOf(survivors) then return "survivor" end
        local spec = wsP:FindFirstChild("Spectating")
        if spec and char:IsDescendantOf(spec) then return "spectator" end
        -- 2) Fallback: ชื่อ char ตรงกับ Model name ใน folder?
        if killers then
            for _, m in pairs(killers:GetChildren()) do
                if m.Name == char.Name then return "killer" end
            end
        end
        if survivors then
            for _, m in pairs(survivors:GetChildren()) do
                if m.Name == char.Name then return "survivor" end
            end
        end
        if spec then
            for _, m in pairs(spec:GetChildren()) do
                if m.Name == char.Name then return "spectator" end
            end
        end
    end
    -- 3) ถ้าไม่มี workspace.Players → ยังอยู่ลอบบี้ → spectator
    return "spectator"
end

local COL_KILLER   = Color3.fromRGB(255,  50,  50)   -- แดง
local COL_SURVIVOR = Color3.fromRGB( 50, 255,  80)   -- เขียว
local COL_SPEC     = Color3.fromRGB(150, 150, 150)   -- เทา — spectator

local function colorFor(role)
    if role == "killer"    then return COL_KILLER   end
    if role == "spectator" then return COL_SPEC     end
    return COL_SURVIVOR
end

local function addHL(p)
    if p == LP or not p.Character then return end
    if Highlights[p.Name] then
        pcall(function() Highlights[p.Name]:Destroy() end)
        Highlights[p.Name] = nil
    end
    local role = detectRole(p, p.Character)
    pCacheRole[p] = role
    local hl = Instance.new("Highlight")
    hl.FillColor           = colorFor(role)
    hl.OutlineColor        = COL_WHITE
    hl.FillTransparency    = 0.5
    hl.OutlineTransparency = 0
    hl.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
    hl.Adornee             = p.Character
    hl.Parent              = p.Character
    hl.Enabled             = ESPOn
    Highlights[p.Name]     = hl
end

local function rebuildColors()
    for i = 1, targetCount do
        local p = activeTargets[i]
        if p.Character then
            local role = detectRole(p, p.Character)
            pCacheRole[p] = role
            local hl = Highlights[p.Name]
            if hl then pcall(function() hl.FillColor = colorFor(role) end) end
        end
    end
end

local function removeHL(name)
    if Highlights[name] then
        pcall(function() Highlights[name]:Destroy() end)
        Highlights[name] = nil
    end
end

-- ==================== Active targets ====================
local function activeAdd(p)
    for i = 1, targetCount do
        if activeTargets[i] == p then return end
    end
    targetCount += 1
    activeTargets[targetCount] = p
end

local function activeRemove(p)
    for i = 1, targetCount do
        if activeTargets[i] == p then
            activeTargets[i] = activeTargets[targetCount]
            activeTargets[targetCount] = nil
            targetCount -= 1
            return
        end
    end
end

local rayParams = RaycastParams.new()

local function updateIgnoreList()
    local chars = {}
    if LPChar then chars[1] = LPChar end
    for i = 1, targetCount do
        local ch = activeTargets[i].Character
        if ch then chars[#chars+1] = ch end
    end
    rayParams.FilterDescendantsInstances = chars
end

local function cacheChar(p, char)
    if not char then
        pCacheHead[p], pCacheRoot[p], pCacheHum[p], pCacheRole[p] = nil, nil, nil, nil
        losCache[p] = nil
        activeRemove(p)
        if lockedPlayer == p then
            lockedPlayer, lockedHead, lockedRoot, lockedHum = nil, nil, nil, nil
        end
        updateIgnoreList()
        return
    end
    pCacheHead[p] = char:FindFirstChild("Head")
    pCacheRoot[p] = char.PrimaryPart or char:FindFirstChild("HumanoidRootPart")
    pCacheHum[p]  = char:FindFirstChildOfClass("Humanoid")
    pCacheRole[p] = detectRole(p, char)
    if pCacheHead[p] and pCacheRoot[p] and pCacheHum[p] then activeAdd(p) end
    updateIgnoreList()
end

local function rescan()
    for name in pairs(Highlights) do removeHL(name) end
    table.clear(activeTargets); targetCount = 0
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LP and p.Character then
            addHL(p)
            cacheChar(p, p.Character)
        end
    end
    print("[FRSN] Rescan:", targetCount, "players")
end

local function setupPlayer(p)

    if p.Character then
        task.spawn(function()
            task.wait(0.4)
            addHL(p)
            cacheChar(p, p.Character)
        end)
    end
    bind(p.CharacterAdded, function(char)
        task.wait(0.4)
        addHL(p)
        cacheChar(p, char)
    end)
    bind(p.CharacterRemoving, function() cacheChar(p, nil) end)
    -- Team change → re-color HL

            pCacheRole[p] = newRole
            local hl = Highlights[p.Name]
            if hl then pcall(function() hl.FillColor = colorFor(newRole) end) end
        end
    end)
end

for _, p in ipairs(Players:GetPlayers()) do setupPlayer(p) end

bind(Players.PlayerRemoving, function(p) removeHL(p.Name); cacheChar(p, nil) end)

        end

        if LPChar then
            MY_ROLE = detectRole(LP, LPChar)
            IS_KILLER = MY_ROLE == "killer"
        end
    end
end)

local function isGenCompleted(genModel)
    -- 1) Progress.Value >= 100 (Forsaken ใช้แบบนี้ — IntValue 0-100)
    local prog = genModel:FindFirstChild("Progress")
    if prog and prog:IsA("IntValue") then return prog.Value >= 100 end
    if prog and prog:IsA("NumberValue") then return prog.Value >= 100 end
    -- 2) BoolValue Completed
    local c = genModel:FindFirstChild("Completed", true)
    if c and c:IsA("BoolValue") then return c.Value end
    -- 3) Attribute
    local a = genModel:GetAttribute("Completed")
    if type(a) == "boolean" then return a end
    -- 4) fallback: Prompt.Enabled
    local p = genModel:FindFirstChildWhichIsA("ProximityPrompt", true)
    if p then return not p.Enabled end
    return false
end

local function scanGenerators()

-- ==================== Find Watch target ====================
-- v2.1: ล็อคเฉพาะ role ตรงข้าม (Killer → Survivor only / Survivor → Killer only)
-- fallback: ถ้า role detect ยังไม่พร้อม (เพิ่งเข้าแมตช์) → เล็งใครก็ได้ใกล้สุด
local function findWatchTarget()
    if not LPRoot then return nil end
    local fromPos = LPRoot.Position
    local myRole = IS_KILLER and "killer" or "survivor"
    local wantRole = IS_KILLER and "survivor" or "killer"

    local bestHead, bestPlayer, bestDSq = nil, nil, math_huge
    local fallbackHead, fallbackPlayer, fallbackDSq = nil, nil, math_huge

    for i = 1, targetCount do
        local p = activeTargets[i]
        local hum = pCacheHum[p]
        if hum and hum.Health > 0 then
            local root = pCacheRoot[p]
            local head = pCacheHead[p]
            if head and root then
                local dx = root.Position.X - fromPos.X
                local dy = root.Position.Y - fromPos.Y
                local dz = root.Position.Z - fromPos.Z
                local dSq = dx*dx + dy*dy + dz*dz

                local role = pCacheRole[p]
                if role == wantRole then
                    -- enemy → priority
                    if dSq < bestDSq then
                        bestDSq, bestHead, bestPlayer = dSq, head, p
                    end
                else
                    -- teammate or unknown → fallback
                    if dSq < fallbackDSq then
                        fallbackDSq, fallbackHead, fallbackPlayer = dSq, head, p
                    end
                end
            end
        end
    end

    if bestHead then
        return bestHead, bestPlayer, math_sqrt(bestDSq)
    end
    -- ไม่เจอ enemy → fallback (กัน role detect ยังไม่ assign)
    if fallbackHead then
        return fallbackHead, fallbackPlayer, math_sqrt(fallbackDSq)
    end
    return nil
end

-- ==================== Movement controller (Climb-hold > Speed) ====================
-- Climb = ไต่กำแพง WASD แต่ใช้ velocity-hold (จาก Fly) → "ไม่ตก" ตอนเกาะกำแพง
-- ไม่มีกำแพงรอบตัว → ปล่อย hold → ร่วงลงปกติ (= ออกจาก Fly อัตโนมัติ)
local UP = v3_new(0, 1, 0)
local climbRayP = RaycastParams.new()
climbRayP.FilterType = Enum.RaycastFilterType.Exclude
local lastClimbVel   = v3_new(0, 0, 0)
local climbHoldUntil = 0   -- ค้าง velocity ต่ออีกนิดตอนข้ามขอบ/ตรวจหลุดแว้บเดียว

-- หากำแพงใกล้สุดรอบตัว (8 ทิศ × 2 ระดับ) → คืน normal แนวนอน (nil = ไม่มีกำแพง)
local function nearestWallNormal()
    climbRayP.FilterDescendantsInstances = LPChar and {LPChar} or {}
    local pos = LPRoot.Position
    local origins = { pos + UP * 1.2, pos - UP * 1.2 }
    local best, bestDist = nil, math_huge
    for i = 0, 7 do
        local a = i * 0.785398   -- 45° step (π/4)
        local d = v3_new(math.cos(a), 0, math.sin(a))
        for _, o in ipairs(origins) do
            local r = workspace:Raycast(o, d * 4.5, climbRayP)
            if r and r.Distance < bestDist then
                local n = v3_new(r.Normal.X, 0, r.Normal.Z)
                if n.Magnitude > 0.01 then bestDist = r.Distance; best = n.Unit end
            end
        end
    end
    return best
end

bind(RunService.Heartbeat, function()
    if not bv or not bv.Parent or not LPHum or LPHum.Health <= 0 then return end

    -- 1) CLIMB (priority) — เกาะกำแพงไม่ตก | W=ขึ้น S=ลง A/D=ไถลข้าง | ปล่อยปุ่ม=ค้างนิ่ง
    -- STEALTH: spoof Humanoid → Climbing state ทุก frame → server เห็นเป็น "ปีนบันได" (legit)
    --   ข้าม heuristic "ลอย+ไม่ตก=บิน" ที่เคยโดนเตะ
    if CLIMB_ON and LPRoot and LPHum then
        local n = nearestWallNormal()
        if n then
            local along = n:Cross(UP)
            along = along.Magnitude > 0.01 and along.Unit or v3_new(0, 0, 0)
            local vel = -n * 4   -- ดันเข้ากำแพง = เกาะติด (ปล่อยปุ่ม = ค้างนิ่ง ไม่ตก)
            if UIS:IsKeyDown(Enum.KeyCode.W) then vel = vel + UP * CLIMB_SPEED end
            if UIS:IsKeyDown(Enum.KeyCode.S) then vel = vel - UP * CLIMB_SPEED end
            if UIS:IsKeyDown(Enum.KeyCode.D) then vel = vel + along * CLIMB_SPEED end
            if UIS:IsKeyDown(Enum.KeyCode.A) then vel = vel - along * CLIMB_SPEED end
            pcall(function() LPHum:ChangeState(Enum.HumanoidStateType.Climbing) end)
            bv.MaxForce = v3_new(1e5, 1e5, 1e5)
            bv.Velocity = vel
            lastClimbVel = vel
            climbHoldUntil = clock() + 0.15   -- grace ข้ามขอบ/ตรวจหลุด
            return
        elseif clock() < climbHoldUntil then
            pcall(function() LPHum:ChangeState(Enum.HumanoidStateType.Climbing) end)
            bv.MaxForce = v3_new(1e5, 1e5, 1e5)
            bv.Velocity = lastClimbVel       -- ค้าง velocity เดิมต่อ (ข้ามขอบ)
            return
        end
        -- ไม่มีกำแพง + หมด grace → ปล่อย hold → ตกปกติ (ออกจาก Fly)
    end

    -- 2) SPEED
    if SPEED_ON then
        local dir = LPHum.MoveDirection
        if dir.Magnitude > 0.1 then
            bv.MaxForce = v3_new(1e5, 0, 1e5)
            bv.Velocity = dir * math_min(SPEED_VAL, SPEED_MAX)
        else
            bv.MaxForce = v3_new(0, 0, 0)
            bv.Velocity = v3_new(0, 0, 0)
        end
    else
        bv.MaxForce = v3_new(0, 0, 0)
        bv.Velocity = v3_new(0, 0, 0)
    end
end)

-- ==================== NOCLIP loop [N] — ทะลุกำแพง STEALTH ====================
-- CanCollide=false ทุก frame (ฝั่ง server ไม่เห็นค่านี้ — ปลอดภัยเอง)
-- สิ่งที่ทำให้รอด: เดินทะลุด้วย WalkSpeed ปกติ (ไม่ warp) + spoof Climbing ทุก frame
--   → server เห็นเป็น "ปีนบันได" ไม่ใช่ "ลอยทะลุของแข็ง" → ข้าม heuristic ที่เคยเตะ climb
-- ปิด → คืน CanCollide เดิมให้ทุก part ทันที
local function setCharCollide(state)
    if not LPChar then return end
    for _, p in pairs(LPChar:GetDescendants()) do
        if p:IsA("BasePart") then
            if state == false then
                if noclipParts[p] == nil then noclipParts[p] = p.CanCollide end
                p.CanCollide = false
            end
        end
    end
end
local function restoreCollide()
    for p, orig in pairs(noclipParts) do
        if p and p.Parent then pcall(function() p.CanCollide = orig end) end
    end
    noclipParts = {}
end

bind(RunService.Heartbeat, function()
    if not NOCLIP_ON then return end
    if not LPChar or not LPHum or LPHum.Health <= 0 then return end
    setCharCollide(false)
    -- STEALTH: spoof ปีนบันได ทุก frame → ตำแหน่งทะลุกำแพงดู legit
    pcall(function() LPHum:ChangeState(Enum.HumanoidStateType.Climbing) end)
    -- cap speed ให้ ≤ 16 ตอน noclip (ห้ามพุ่ง position-delta = โดน flag)
    if LPHum.WalkSpeed > 16 then LPHum.WalkSpeed = 16 end
end)

-- ==================== AUTO-GEN loop (Survivor only) — NO TELEPORT ====================
-- ห้ามวาป! เกมแบนถ้า server detect CFrame jump
-- → script ทำแค่ fire prompt อัตโนมัติเมื่อผู้เล่นเดินมาใกล้ ≤ AUTOGEN_DIST
-- → ใช้ ESP gen (cyan highlight) ช่วยมอง → เดินไปเอง → script กด E ค้างให้
local AUTOGEN_DIST = 8   -- ระยะ ≤ นี้ studs ค่อย fire prompt (auto-press E)
local AUTOGEN_FIRE_COUNT = 6     -- จำนวน RE:FireServer ต่อ session
local AUTOGEN_FIRE_GAP   = 2.5   -- gap ระหว่าง fire (ห้าม < 2.5 → kick)
local autogenStatus = "idle"
local autogenBusy = {}           -- [genModel] = true (กำลังยิง progress อยู่)

local function pushGenProgress(g)
    if autogenBusy[g.model] then return end
    autogenBusy[g.model] = true
    task.spawn(function()
        -- patch stamina ครั้งเดียว
        pcall(function()
            require(game.ReplicatedStorage.Systems.Character.Game.Sprinting)
                .StaminaLossDisabled = true
        end)
        -- กด E ก่อน (เปิด puzzle session)
        pcall(function() fireproximityprompt(g.prompt) end)
        task.wait(0.4)
        -- ยิง progress 6 ครั้ง
        local RE = g.model:FindFirstChild("Remotes")
        RE = RE and RE:FindFirstChild("RE")
        if RE then
            for i = 1, AUTOGEN_FIRE_COUNT do
                if not AUTOGEN_ON then break end
                if not g.model or not g.model.Parent then break end
                if isGenCompleted(g.model) then break end
                pcall(function() RE:FireServer() end)
                task.wait(AUTOGEN_FIRE_GAP)
            end
        end
        autogenBusy[g.model] = nil
    end)
end

-- ==================== STALK (Survivor: walk to behind Killer) ====================
-- ใช้ Humanoid:Move() = เดินปกติ + animation play เอง = ไม่ teleport = ปลอดภัย
-- ระยะหลัง Killer ปรับใน GUI | หลังเป็นกำแพง → ลอง Right → ลอง Left
local stalkStatus   = "idle"
local origWalkSpeed = nil   -- เก็บ WalkSpeed เดิม restore ตอน OFF

-- raycast เช็คกำแพงระหว่าง 2 จุด (exclude characters + hitboxes)
local stalkRayParams = RaycastParams.new()
stalkRayParams.FilterType = Enum.RaycastFilterType.Exclude
local function refreshStalkIgnore()
    local list = {}
    if LPChar then table.insert(list, LPChar) end
    for i = 1, targetCount do
        local ch = activeTargets[i].Character
        if ch then table.insert(list, ch) end
    end
    -- exclude Hitboxes folder (custom hitbox ของเกม)
    local hb = workspace:FindFirstChild("Hitboxes")
    if hb then table.insert(list, hb) end
    stalkRayParams.FilterDescendantsInstances = list
end

local function hasWallStalk(from, to)
    local dir = to - from
    if dir.Magnitude < 0.1 then return false end
    local r = workspace:Raycast(from + v3_new(0, 2, 0), dir, stalkRayParams)
    return r ~= nil
end

-- หา position ปลอดภัย "หลัง" Killer (ลอง back → right → left)
local function findStalkSpot(killerRoot)
    local kp   = killerRoot.Position
    local look = killerRoot.CFrame.LookVector
    local right = killerRoot.CFrame.RightVector
    local d = effStalkDist()    -- 2 ถ้า Killer, 22 ถ้า Survivor
    -- candidates เรียงจาก preferred → fallback
    local candidates = {
        { pos = kp - look * d,  label = "back" },
        { pos = kp + right * d, label = "right" },
        { pos = kp - right * d, label = "left" },
        { pos = kp + look * d,  label = "front" },  -- last resort
    }
    for _, c in ipairs(candidates) do
        local target = v3_new(c.pos.X, kp.Y, c.pos.Z)
        if not hasWallStalk(kp, target) then
            return target, c.label
        end
    end
    return nil, "blocked"
end

bind(RunService.Heartbeat, function()
    if not STALK_ON then
        if origWalkSpeed and LPHum then
            pcall(function() LPHum.WalkSpeed = origWalkSpeed end)
            origWalkSpeed = nil
        end
        stalkStatus = "off"
        return
    end
    if not LPRoot or not LPHum or LPHum.Health <= 0 then return end

    if not origWalkSpeed then origWalkSpeed = LPHum.WalkSpeed end
    if LPHum.WalkSpeed ~= STALK_SPEED then
        pcall(function() LPHum.WalkSpeed = STALK_SPEED end)
    end

    refreshStalkIgnore()
    local fromPos = LPRoot.Position
    local nearestRoot, nearestD = nil, math_huge
    for i = 1, targetCount do
        local p = activeTargets[i]
        local hum = pCacheHum[p]
        if hum and hum.Health > 0 then
            local root = pCacheRoot[p]
            if root then
                local d = (root.Position - fromPos).Magnitude
                if d < nearestD then nearestD = d nearestRoot = root end
            end
        end
    end

    if not nearestRoot or nearestD > STALK_TRIGGER then
        stalkStatus = string.format("idle (no enemy ≤ %d)", STALK_TRIGGER)
        return
    end

    local target, label = findStalkSpot(nearestRoot)
    if not target then
        stalkStatus = "all blocked"
        return
    end
    local dir = target - fromPos
    dir = v3_new(dir.X, 0, dir.Z)
    if dir.Magnitude > 0.5 then
        LPHum:Move(dir.Unit, false)
        stalkStatus = string.format("→ %s d=%.0f", label, nearestD)
    else
        LPHum:Move(v3_new(0, 0, 0), false)
        stalkStatus = string.format("✓ %s d=%.0f", label, nearestD)
    end
end)

local function toggleESP()
    ESPOn = not ESPOn
    setAllEnabled(ESPOn)
    espBtn.BackgroundColor3 = ESPOn and Color3.fromRGB(180, 30, 100) or Color3.fromRGB(50, 50, 50)
    espBtn.Text = "ESP " .. (ESPOn and "ON" or "OFF") .. " [H]"

local function updS()
    sVal.Text = tostring(SPEED_VAL)

        sTog.Text = "Speed OFF [F]"

local function adjK(delta)
    if IS_KILLER then
        STALK_DIST_KILLER = math_clamp(STALK_DIST_KILLER + delta, 1, 60)
    else
        STALK_DIST_SURVIVOR = math_clamp(STALK_DIST_SURVIVOR + delta, 1, 60)
    end
    updK()
end

bind(UIS.InputBegan, function(input, gp)
    if gp then return end
    local kc = input.KeyCode
    if     kc == Enum.KeyCode.H then toggleESP()
    elseif kc == Enum.KeyCode.T then toggleWatch()
    elseif kc == Enum.KeyCode.G then toggleAutoGen()
    elseif kc == Enum.KeyCode.F then togS()
    elseif kc == Enum.KeyCode.K then togK()
    elseif kc == Enum.KeyCode.C then
        CLIMB_ON = not CLIMB_ON
    elseif kc == Enum.KeyCode.N then
        NOCLIP_ON = not NOCLIP_ON
        if not NOCLIP_ON then restoreCollide() end
    elseif kc == Enum.KeyCode.R then
        rescan(); scanGenerators(); rebuildGenHL(); scanItems(); rebuildItemHL()
        rescanBtn.Text = "Done!"
        task.delay(1.2, function() rescanBtn.Text = "Rescan [R]" end)
    end
end)

task.spawn(function()
    while task.wait(1) do
        if not _G.FRSN_CONNS or _G.FRSN_CONNS ~= CONNS then return end
        local done = 0
        for _, g in pairs(cachedGens) do
            if g.model and g.model.Parent and isGenCompleted(g.model) then done += 1 end
        end
        statL.Text = "P:" .. targetCount .. " G:" .. done .. "/" .. #cachedGens
            .. " I:" .. #cachedItems
        if WatchOn then
            statL.Text = statL.Text .. " | 👁️ " .. watchTargetName
                .. " " .. math.floor(watchTargetDist) .. "m"
        end
        if AUTOGEN_ON then statL.Text = statL.Text .. " | " .. autogenStatus end
        if STALK_ON then statL.Text = statL.Text .. " | 🥷 " .. stalkStatus end
        if CLIMB_ON  then statL.Text = statL.Text .. " | 🧗Climb" end
        if NOCLIP_ON then statL.Text = statL.Text .. " | 👻Noclip" end
        if MY_ROLE == "killer" then
            roleL.Text = "Role: 🔪 KILLER"
            roleL.TextColor3 = Color3.fromRGB(255, 100, 100)
        elseif MY_ROLE == "survivor" then
            roleL.Text = "Role: 🟢 SURVIVOR"
            roleL.TextColor3 = Color3.fromRGB(100, 255, 130)
        else
            roleL.Text = "Role: 👁️ SPECTATOR"
            roleL.TextColor3 = Color3.fromRGB(180, 180, 180)
        end
        updK()   -- refresh stalk distance display (เปลี่ยนตาม role)
        -- update gen highlights (เสร็จ → DESTROY ทิ้ง)
        if #GenHL > 0 then
            for i, g in pairs(cachedGens) do
                local hl = GenHL[i]
                if hl and hl.Parent then
                    if isGenCompleted(g.model) then
                        pcall(function() hl:Destroy() end)
                        GenHL[i] = nil
                    elseif ESPOn then
                        hl.FillColor = COL_GEN_TODO
                    end
                end
            end
        end
    end
end)

-- ==================== ROLE DEBUG ====================
-- print Team + Attribute ของผู้เล่นทุกคน → ใช้ปรับ detect logic
local function debugRoles()
    print("════════ [FRSN ROLE DEBUG] ════════")
    -- 1. Teams service
    local TeamsSvc = game:GetService("Teams")
    local teams = TeamsSvc:GetTeams()
    print("Teams in service:", #teams)
    for i, t in ipairs(teams) do
        print(string.format("  [%d] '%s' color=%s", i, t.Name, tostring(t.TeamColor)))
    end
    -- 2. ทุก player → Team / Attributes / Char name
    print("Players:")
    for _, p in ipairs(Players:GetPlayers()) do
        local tm = p.Team and p.Team.Name or "(no team)"
        local chName = p.Character and p.Character.Name or "(no char)"
        local attrStr = ""
        for k, v in pairs(p:GetAttributes()) do
            attrStr = attrStr .. k .. "=" .. tostring(v) .. " "
        end
        local me = (p == LP) and " [ME]" or ""
        print(string.format("  %s%s | Team=%s | Char=%s | Attr={%s}",
            p.Name, me, tm, chName, attrStr))
    end
    print("LP.Character attributes:")
    if LPChar then
        for k, v in pairs(LPChar:GetAttributes()) do
            print("  Char attr:", k, "=", v)
        end
        if LPHum then
            print("  Hum attributes:")
            for k, v in pairs(LPHum:GetAttributes()) do
                print("    ", k, "=", v)
            end
        end
    end
    print("══════════════════════════════════")
end
debugRoles()

-- ==================== Role detect (Forsaken pattern) ====================
-- Char.Name == Player.Name → Survivor (Survivor outfit ใช้ display name)
-- Char.Name ~= Player.Name → Killer  (Slasher/Jason/007n7/1x1x1x1/etc.)
local function detectRole(player, char)
    if not char then return "survivor" end
    if char.Name == player.Name then return "survivor" end
    return "killer"
end

-- ==================== Highlight (ESP) ====================
local COL_KILLER   = Color3.fromRGB(255,  50,  50)   -- แดง
local COL_SURVIVOR = Color3.fromRGB( 50, 255,  80)   -- เขียว
local COL_GEN_TODO = Color3.fromRGB(  0, 200, 255)   -- ฟ้า — gen ยังไม่เสร็จ
local COL_GEN_DONE = Color3.fromRGB(120, 120, 120)   -- เทา — gen เสร็จแล้ว
local COL_WHITE    = Color3.fromRGB(255, 255, 255)

local function colorFor(role)
    return role == "killer" and COL_KILLER or COL_SURVIVOR
end

local function addHL(p)
    if p == LP or not p.Character then return end
    if Highlights[p.Name] then
        pcall(function() Highlights[p.Name]:Destroy() end)
        Highlights[p.Name] = nil
    end
    local role = detectRole(p, p.Character)
    pCacheRole[p] = role
    local hl = Instance.new("Highlight")
    hl.FillColor           = colorFor(role)
    hl.OutlineColor        = COL_WHITE
    hl.FillTransparency    = 0.5
    hl.OutlineTransparency = 0
    hl.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
    hl.Adornee             = p.Character
    hl.Parent              = p.Character
    hl.Enabled             = ESPOn
    Highlights[p.Name]     = hl
end

local function rebuildColors()
    for i = 1, targetCount do
        local p = activeTargets[i]
        if p.Character then
            local role = detectRole(p, p.Character)
            pCacheRole[p] = role
            local hl = Highlights[p.Name]
            if hl then pcall(function() hl.FillColor = colorFor(role) end) end
        end
    end
end

local function removeHL(name)
    if Highlights[name] then
        pcall(function() Highlights[name]:Destroy() end)
        Highlights[name] = nil
    end
end

local function setAllEnabled(val)
    for _, hl in pairs(Highlights) do
        pcall(function() hl.Enabled = val end)
    end
    for _, hl in pairs(GenHL) do
        pcall(function() hl.Enabled = val end)
    end
end

-- ==================== Active targets ====================
local function activeAdd(p)
    for i = 1, targetCount do
        if activeTargets[i] == p then return end
    end
    targetCount += 1
    activeTargets[targetCount] = p
end

local function activeRemove(p)
    for i = 1, targetCount do
        if activeTargets[i] == p then
            activeTargets[i] = activeTargets[targetCount]
            activeTargets[targetCount] = nil
            targetCount -= 1
            return
        end
    end
end

local rayParams = RaycastParams.new()
rayParams.FilterType = Enum.RaycastFilterType.Exclude
local function updateIgnoreList()
    local chars = {}
    if LPChar then chars[1] = LPChar end
    for i = 1, targetCount do
        local ch = activeTargets[i].Character
        if ch then chars[#chars+1] = ch end
    end
    rayParams.FilterDescendantsInstances = chars
end

local function cacheChar(p, char)
    if not char then
        pCacheHead[p], pCacheRoot[p], pCacheHum[p], pCacheRole[p] = nil, nil, nil, nil
        losCache[p] = nil
        activeRemove(p)
        if lockedPlayer == p then
            lockedPlayer, lockedHead, lockedRoot, lockedHum = nil, nil, nil, nil
        end
        updateIgnoreList()
        return
    end
    pCacheHead[p] = char:FindFirstChild("Head")
    pCacheRoot[p] = char.PrimaryPart or char:FindFirstChild("HumanoidRootPart")
    pCacheHum[p]  = char:FindFirstChildOfClass("Humanoid")
    pCacheRole[p] = detectRole(p, char)
    if pCacheHead[p] and pCacheRoot[p] and pCacheHum[p] then activeAdd(p) end
    updateIgnoreList()
end

local function rescan()
    for name in pairs(Highlights) do removeHL(name) end
    table.clear(activeTargets); targetCount = 0
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LP and p.Character then
            addHL(p)
            cacheChar(p, p.Character)
        end
    end
    print("[FRSN] Rescan:", targetCount, "players")
end

local function setupPlayer(p)
    if p == LP then return end
    if p.Character then
        task.spawn(function()
            task.wait(0.4)
            addHL(p)
            cacheChar(p, p.Character)
        end)
    end
    bind(p.CharacterAdded, function(char)
        task.wait(0.4)
        addHL(p)
        cacheChar(p, char)
    end)
    bind(p.CharacterRemoving, function() cacheChar(p, nil) end)
end

-- Periodic re-check (heal lost highlights + role change)
task.spawn(function()
    while task.wait(2) do
        if not _G.FRSN_CONNS or _G.FRSN_CONNS ~= CONNS then return end
        if not ESPOn then continue end
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LP and p.Character then
                local hl = Highlights[p.Name]
                if not hl or not hl.Parent then addHL(p) end
                local h = pCacheHead[p]
                if not h or not h.Parent then cacheChar(p, p.Character) end
                -- role-change detect (เปลี่ยน char glow → update สี)
                local newRole = detectRole(p, p.Character)
                if pCacheRole[p] ~= newRole then
                    pCacheRole[p] = newRole
                    if hl then pcall(function() hl.FillColor = colorFor(newRole) end) end
                end
            end
        end
        -- ตรวจ role ของเราเอง
        if LPChar then IS_KILLER = detectRole(LP, LPChar) == "killer" end
    end
end)

-- ==================== GENERATOR scan + ESP ====================
local cachedGens = {}    -- list ของ { model, prompt, part, completed? }
local lastGenScan = 0

local function isGenCompleted(genModel)
    -- ลอง BoolValue Completed
    local c = genModel:FindFirstChild("Completed", true)
    if c and c:IsA("BoolValue") then return c.Value end
    -- ลอง Attribute
    local a = genModel:GetAttribute("Completed")
    if type(a) == "boolean" then return a end
    -- fallback: ดู Prompt.Enabled (เสร็จ → disabled)
    local p = genModel:FindFirstChildWhichIsA("ProximityPrompt", true)
    if p then return not p.Enabled end
    return false
end

local function scanGenerators()
    -- path หลัก: Workspace.Map.Ingame.Map.Generator (หรือ Lobby.Interactive.Generator)
    cachedGens = {}
    local function tryScan(parent)
        if not parent then return end
        for _, ch in pairs(parent:GetChildren()) do
            if ch.Name == "Generator" and ch:IsA("Model") then
                local prompt = ch:FindFirstChildWhichIsA("ProximityPrompt", true)
                local part   = ch:FindFirstChildWhichIsA("BasePart", true)
                if prompt and part then
                    table.insert(cachedGens, { model = ch, prompt = prompt, part = part })
                end
            end
        end
    end
    pcall(function()
        local map = workspace:FindFirstChild("Map")
        if map then
            local ig = map:FindFirstChild("Ingame")
            if ig then
                local m2 = ig:FindFirstChild("Map")
                if m2 then tryScan(m2) end
            end
            local lobby = map:FindFirstChild("Lobby")
            if lobby then
                local int = lobby:FindFirstChild("Interactive")
                if int then tryScan(int) end
            end
        end
    end)
end

local function rebuildGenHL()
    for _, hl in pairs(GenHL) do pcall(function() hl:Destroy() end) end
    table.clear(GenHL)
    if not ESPOn then return end
    for _, g in pairs(cachedGens) do
        if g.model and g.model.Parent then
            local hl = Instance.new("Highlight")
            hl.FillColor    = isGenCompleted(g.model) and COL_GEN_DONE or COL_GEN_TODO
            hl.OutlineColor = COL_WHITE
            hl.FillTransparency    = 0.7
            hl.OutlineTransparency = 0.2
            hl.DepthMode    = Enum.HighlightDepthMode.AlwaysOnTop
            hl.Adornee      = g.model
            hl.Parent       = g.model
            hl.Enabled      = ESPOn
            table.insert(GenHL, hl)
        end
    end
end

-- ==================== Find Watch target ====================
-- v1.3: เลิกกรอง role (detect ไม่แม่น) → เล็งใครก็ได้ใกล้สุดที่ไม่ใช่ตัวเอง
-- Killer: ใกล้สุดมักเป็น Survivor (Killer ตัวเดียว)
-- Survivor: ใกล้สุดมักเป็น Killer (ถ้า killer ตามมา)
local function findWatchTarget()
    if not LPRoot then return nil end
    local fromPos = LPRoot.Position
    local bestHead, bestPlayer, bestDSq = nil, nil, math_huge
    for i = 1, targetCount do
        local p = activeTargets[i]
        local hum = pCacheHum[p]
        if hum and hum.Health > 0 then
            local root = pCacheRoot[p]
            local head = pCacheHead[p]
            if head and root then
                local dx = root.Position.X - fromPos.X
                local dy = root.Position.Y - fromPos.Y
                local dz = root.Position.Z - fromPos.Z
                local dSq = dx*dx + dy*dy + dz*dz
                if dSq < bestDSq then
                    bestDSq, bestHead, bestPlayer = dSq, head, p
                end
            end
        end
    end
    return bestHead, bestPlayer, math_sqrt(bestDSq)
end

-- ==================== Watch (BindToRenderStep — priority Last) ====================
-- v1.3: Camera.Value+1 (201) ไม่พอ — Forsaken อาจ override ใน Character (300)
-- → ใช้ Last (2000) บังคับรันท้ายสุดของ render frame → เขียนทับเกมได้ชัวร์
local watchTargetName = "-"
local watchTargetDist = 0
local WATCH_BIND = "FRSN_Watch_Bind"
pcall(function() RunService:UnbindFromRenderStep(WATCH_BIND) end)
RunService:BindToRenderStep(WATCH_BIND, Enum.RenderPriority.Last.Value, function()
    if not WatchOn or not LPRoot then return end
    local head, player, dist = findWatchTarget()
    if head then
        Camera.CFrame = cf_lookAt(Camera.CFrame.Position, head.Position)
        watchTargetName = player and player.Name or "?"
        watchTargetDist = dist or 0
    else
        watchTargetName = "(no target)"
        watchTargetDist = 0
    end
end)
table.insert(CONNS, { Disconnect = function()
    pcall(function() RunService:UnbindFromRenderStep(WATCH_BIND) end)
end })

-- ==================== AUTO-GEN loop (Survivor only) — NO TELEPORT ====================
-- ห้ามวาป! เกมแบนถ้า server detect CFrame jump
-- → script ทำแค่ fire prompt อัตโนมัติเมื่อผู้เล่นเดินมาใกล้ ≤ AUTOGEN_DIST
-- → ใช้ ESP gen (cyan highlight) ช่วยมอง → เดินไปเอง → script กด E ค้างให้
local AUTOGEN_DIST = 8   -- ระยะ ≤ นี้ studs ค่อย fire prompt (auto-press E)
local autogenStatus = "idle"
bind(RunService.Heartbeat, function()
    if not AUTOGEN_ON then autogenStatus = "off" return end
    if IS_KILLER then autogenStatus = "killer (skip)" return end
    if not LPRoot or not LPHum or LPHum.Health <= 0 then return end
    GEN_TICK += 1
    if GEN_TICK % 3 ~= 0 then return end   -- ~20/วิ (fire prompt บ่อยพอสำหรับ hold)

    -- rescan gen ทุก 5 วิ (เผื่อเข้า match ใหม่)
    if clock() - lastGenScan > 5 then
        scanGenerators()
        rebuildGenHL()
        lastGenScan = clock()
    end

    -- หา gen ใกล้สุดที่ยังไม่เสร็จ + อยู่ในระยะ
    local fromPos = LPRoot.Position
    local best, bestD = nil, math_huge
    for _, g in pairs(cachedGens) do
        if g.model and g.model.Parent and not isGenCompleted(g.model) then
            local d = (g.part.Position - fromPos).Magnitude
            if d < bestD then bestD = d best = g end
        end
    end
    if not best then autogenStatus = "no gen" return end

    if bestD <= AUTOGEN_DIST then
        -- ใกล้พอ → fire prompt (กดค้าง E แทนเรา)
        pcall(function() fireproximityprompt(best.prompt) end)
        autogenStatus = string.format("✓ ปั่น (%.1fm)", bestD)
    else
        -- ยังไกล → แสดงระยะ + ทิศ (ESP cyan ช่วยมอง)
        autogenStatus = string.format("เดินไป Gen %.0fm", bestD)
    end
end)

-- ==================== STALK (Survivor: walk to behind Killer) ====================
-- ทำงานเสมอเมื่อเปิด (ไม่กรอง role เพราะ detect อาจผิด)
-- ใช้ Humanoid:Move() = เดินปกติ + animation play เอง = ไม่ teleport = ปลอดภัย
-- ระยะหลัง Killer ปรับใน GUI | หลังเป็นกำแพง → ลอง Right → ลอง Left
local stalkStatus   = "idle"
local origWalkSpeed = nil   -- เก็บ WalkSpeed เดิม restore ตอน OFF

-- raycast เช็คกำแพงระหว่าง 2 จุด (exclude characters + hitboxes)
local stalkRayParams = RaycastParams.new()
stalkRayParams.FilterType = Enum.RaycastFilterType.Exclude
local function refreshStalkIgnore()
    local list = {}
    if LPChar then table.insert(list, LPChar) end
    for i = 1, targetCount do
        local ch = activeTargets[i].Character
        if ch then table.insert(list, ch) end
    end
    -- exclude Hitboxes folder (custom hitbox ของเกม)
    local hb = workspace:FindFirstChild("Hitboxes")
    if hb then table.insert(list, hb) end
    stalkRayParams.FilterDescendantsInstances = list
end

local function hasWallStalk(from, to)
    local dir = to - from
    if dir.Magnitude < 0.1 then return false end
    local r = workspace:Raycast(from + v3_new(0, 2, 0), dir, stalkRayParams)
    return r ~= nil
end

-- หา position ปลอดภัย "หลัง" Killer (ลอง back → right → left)
local function findStalkSpot(killerRoot)
    local kp   = killerRoot.Position
    local look = killerRoot.CFrame.LookVector
    local right = killerRoot.CFrame.RightVector
    -- candidates เรียงจาก preferred → fallback
    local candidates = {
        { pos = kp - look * STALK_DIST,  label = "back" },
        { pos = kp + right * STALK_DIST, label = "right" },
        { pos = kp - right * STALK_DIST, label = "left" },
        { pos = kp + look * STALK_DIST,  label = "front" },  -- last resort
    }
    for _, c in ipairs(candidates) do
        local target = v3_new(c.pos.X, kp.Y, c.pos.Z)
        if not hasWallStalk(kp, target) then
            return target, c.label
        end
    end
    return nil, "blocked"
end

bind(RunService.Heartbeat, function()
    if not STALK_ON then
        -- restore WalkSpeed ตอน OFF
        if origWalkSpeed and LPHum then
            pcall(function() LPHum.WalkSpeed = origWalkSpeed end)
            origWalkSpeed = nil
        end
        stalkStatus = "off"
        return
    end
    if not LPRoot or not LPHum or LPHum.Health <= 0 then return end

    -- บันทึก WalkSpeed เดิมครั้งแรก + set max 29
    if not origWalkSpeed then origWalkSpeed = LPHum.WalkSpeed end
    if LPHum.WalkSpeed ~= STALK_SPEED then
        pcall(function() LPHum.WalkSpeed = STALK_SPEED end)
    end

    -- หา player ใกล้สุด (สมมติเป็น Killer ที่ตามมา)
    refreshStalkIgnore()
    local fromPos = LPRoot.Position
    local nearestRoot, nearestD = nil, math_huge
    for i = 1, targetCount do
        local p = activeTargets[i]
        local hum = pCacheHum[p]
        if hum and hum.Health > 0 then
            local root = pCacheRoot[p]
            if root then
                local d = (root.Position - fromPos).Magnitude
                if d < nearestD then nearestD = d nearestRoot = root end
            end
        end
    end

    if not nearestRoot or nearestD > STALK_TRIGGER then
        stalkStatus = string.format("idle (no enemy ≤ %d)", STALK_TRIGGER)
        return
    end

    -- หาจุดยืนหลัง Killer (หลีกกำแพง)
    local target, label = findStalkSpot(nearestRoot)
    if not target then
        stalkStatus = "all blocked"
        return
    end

    -- เดินไปจุด target ผ่าน Humanoid:Move (animation play เอง, ไม่ teleport)
    local dir = target - fromPos
    dir = v3_new(dir.X, 0, dir.Z)
    if dir.Magnitude > 0.5 then
        LPHum:Move(dir.Unit, false)
        stalkStatus = string.format("→ %s d=%.0f", label, nearestD)
    else
        -- ถึงแล้ว → หยุดเดิน
        LPHum:Move(v3_new(0, 0, 0), false)
        stalkStatus = string.format("✓ %s d=%.0f", label, nearestD)
    end
end)

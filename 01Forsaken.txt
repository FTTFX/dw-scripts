-- ==================== Highlight ====================
local COL_ENEMY    = Color3.fromRGB(255, 50, 50)
local COL_TEAMMATE = Color3.fromRGB(50, 255, 80)
local COL_WHITE    = Color3.fromRGB(255, 255, 255)

local function getColor(p, myID)
    if TeamCheck and myID and getTeamID(p) == myID then return COL_TEAMMATE end
    return COL_ENEMY
end

local function addHL(p)
    if p == LP or not p.Character then return end
    if Highlights[p.Name] then
        pcall(function() Highlights[p.Name]:Destroy() end)
        Highlights[p.Name] = nil
    end
    local myID = TeamCheck and getTeamID(LP) or nil
    local hl = Instance.new("Highlight")
    hl.FillColor           = getColor(p, myID)
    hl.OutlineColor        = COL_WHITE
    hl.FillTransparency    = 0.5
    hl.OutlineTransparency = 0
    hl.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
    hl.Adornee             = p.Character
    hl.Parent              = p.Character
    hl.Enabled             = ESPOn
    Highlights[p.Name]     = hl
end

local function rebuildColors()
    local myID = TeamCheck and getTeamID(LP) or nil
    for i = 1, targetCount do
        local p = activeTargets[i]
        local hl = Highlights[p.Name]
        if hl then pcall(function() hl.FillColor = getColor(p, myID) end) end
    end
end

local function removeHL(name)
    if Highlights[name] then
        pcall(function() Highlights[name]:Destroy() end)
        Highlights[name] = nil
    end
end

local function setAllEnabled(val)
    for _, hl in pairs(Highlights) do
        pcall(function() hl.Enabled = val end)
    end
end

-- ==================== Active targets incremental ====================
local function activeAdd(p)
    for i = 1, targetCount do
        if activeTargets[i] == p then return end
    end
    targetCount = targetCount + 1
    activeTargets[targetCount] = p
end

local function activeRemove(p)
    for i = 1, targetCount do
        if activeTargets[i] == p then
            activeTargets[i] = activeTargets[targetCount]
            activeTargets[targetCount] = nil
            targetCount = targetCount - 1
            return
        end
    end
end

-- ==================== Raycast filter ====================
local rayParams = RaycastParams.new()
rayParams.FilterType = Enum.RaycastFilterType.Exclude

local function updateIgnoreList()
    local chars = {}
    if LPChar then chars[1] = LPChar end
    for i = 1, targetCount do
        local ch = activeTargets[i].Character
        if ch then chars[#chars+1] = ch end
    end
    rayParams.FilterDescendantsInstances = chars
end

-- ==================== Shield Detection ====================
-- raycast ชุดนี้ exclude แค่ตัวเราเอง → เจอ part ของศัตรู (รวม tool/shield) ได้
local shieldRayParams = RaycastParams.new()
shieldRayParams.FilterType = Enum.RaycastFilterType.Exclude

local SHIELD_KEYWORDS = {"shield", "barrier", "wall", "block", "guard", "riot"}

local function isShieldPart(inst)
    if not inst then return false end
    local name = inst.Name:lower()
    for _, kw in ipairs(SHIELD_KEYWORDS) do
        if name:find(kw) then return true end
    end
    -- เช็ค Tool ancestor (ถ้าโล่เป็น Tool)
    local tool = inst:FindFirstAncestorOfClass("Tool")
    if tool then
        local tname = tool.Name:lower()
        for _, kw in ipairs(SHIELD_KEYWORDS) do
            if tname:find(kw) then return true end
        end
    end
    return false
end

local function isShielded(fromPos, headPos)
    local result = World:Raycast(fromPos, headPos - fromPos, shieldRayParams)
    if not result then return false end          -- path โล่ง
    if isShieldPart(result.Instance) then
        return true                              -- โดนโล่งดัก
    end
    return false  -- โดนกำแพง/พื้น = แค่ไม่มี LOS ปกติ (ไม่ใช่โล่)
end

function updateShieldIgnoreList()
    shieldRayParams.FilterDescendantsInstances = LPChar and {LPChar} or {}
end

-- ==================== Player Cache ====================
local function cacheChar(p, char)
    if not char then
        pCacheHead[p] = nil
        pCacheRoot[p] = nil
        pCacheHum[p]  = nil
        losCache[p]   = nil
        activeRemove(p)
        if lockedPlayer == p then
            lockedPlayer, lockedHead, lockedRoot, lockedHum = nil, nil, nil, nil
        end
        updateIgnoreList()
        return
    end
    local head = char:FindFirstChild("Head")
    local root = char.PrimaryPart or char:FindFirstChild("HumanoidRootPart")
    local hum  = char:FindFirstChildOfClass("Humanoid")
    pCacheHead[p] = head
    pCacheRoot[p] = root
    pCacheHum[p]  = hum
    if head and root and hum then activeAdd(p) end
    updateIgnoreList()
end

local function rescan()
    for name in pairs(Highlights) do removeHL(name) end
    table.clear(activeTargets); targetCount = 0
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LP and p.Character then
            addHL(p)
            cacheChar(p, p.Character)
        end
    end
    print("[06RB v3.2] Rescan:", targetCount, "players")
end

-- ==================== Predict + PreAim ====================
local function predictHead(headPos, velX, velY, velZ, fromPos)
    if NETCODE_MODE == "CLIENT_HITREG" or NETCODE_MODE == "LAG_COMP" then
        return headPos
    end
    local ping = 0
    local ok, p = pcall(LP.GetNetworkPing, LP)
    if ok and p then ping = p end
    local travelTime = ping * 2
    if NETCODE_MODE == "SERVER_PROJECTILE" then
        travelTime = travelTime + (fromPos - headPos).Magnitude / BULLET_SPEED
    end
    return headPos + v3_new(velX * travelTime, velY * 0.3 * travelTime, velZ * travelTime)
end

local function getPreAimPosition(fromPos, headPos, velX, velZ)
    local speedSq = velX*velX + velZ*velZ
    if speedSq < 4 then return headPos end
    local speed = math_sqrt(speedSq)
    local step  = math_clamp(1 / speed, 0.04, 0.1)
    local maxT  = math_min(0.8, step * PREAIM_MAX_ITER)
    local flatVel = v3_new(velX, 0, velZ)
    for t = step, maxT, step do
        local checkPos = headPos + flatVel * t
        if not World:Raycast(fromPos, checkPos - fromPos, rayParams) then
            return checkPos
        end
    end
    return headPos
end

local frameCounter = 0

-- ==================== Target scan ====================
local function findBestTarget(refreshLOS)
    if not LPRoot then return nil end
    local fromPos = LPRoot.Position
    local fpx, fpy, fpz = fromPos.X, fromPos.Y, fromPos.Z
    local myID = TeamCheck and getTeamID(LP) or nil

    if lockedPlayer and lockedHum and lockedHum.Health > 0 and lockedHead and lockedRoot then
        local rp = lockedRoot.Position
        local dx, dy, dz = rp.X - fpx, rp.Y - fpy, rp.Z - fpz
        if dx*dx + dy*dy + dz*dz <= SHOOT_MAX_DIST_SQ then
            local hp = lockedHead.Position
            if not World:Raycast(fromPos, hp - fromPos, rayParams) then
                losCache[lockedPlayer] = true
                -- LOS ยังอยู่ → คง lock ไว้ (ไม่สวิตช์แม้มีคนอื่นใกล้กว่า)
                return lockedHead, true, lockedHead.AssemblyLinearVelocity, lockedPlayer
            end
            -- เป้าหลบแล้ว → clear lock → สวิตช์ไปคนถัดไปที่มี LOS
        end
        lockedPlayer, lockedHead, lockedRoot, lockedHum = nil, nil, nil, nil
    end

    local bestLOS_h, bestLOS_v, bestLOS_p
    local bestAny_h, bestAny_v, bestAny_p
    local bestLOS_dSq = SHOOT_MAX_DIST_SQ + 1
    local bestAny_dSq = math_huge

    for i = 1, targetCount do
        local p = activeTargets[i]
        local hum = pCacheHum[p]
        if hum and hum.Health > 0 and not (myID and getTeamID(p) == myID) then
            local root = pCacheRoot[p]
            local head = pCacheHead[p]
            local rp = root.Position
            local dx, dy, dz = rp.X - fpx, rp.Y - fpy, rp.Z - fpz
            local dSq = dx*dx + dy*dy + dz*dz

            if dSq < bestAny_dSq then
                bestAny_dSq, bestAny_h, bestAny_v, bestAny_p =
                    dSq, head, head.AssemblyLinearVelocity, p
            end

            if dSq < bestLOS_dSq and dSq <= SHOOT_MAX_DIST_SQ then
                local hasLOS
                if refreshLOS or losCache[p] == nil then
                    local hp = head.Position
                    hasLOS = World:Raycast(fromPos, hp - fromPos, rayParams) == nil
                    losCache[p] = hasLOS
                else
                    hasLOS = losCache[p]
                end
                if hasLOS then
                    bestLOS_dSq, bestLOS_h, bestLOS_v, bestLOS_p =
                        dSq, head, head.AssemblyLinearVelocity, p
                end
            end
        end
    end

    if bestLOS_h then return bestLOS_h, true,  bestLOS_v, bestLOS_p end
    return            bestAny_h, false, bestAny_v, bestAny_p
end

_G.RB06_FIND = function() return (findBestTarget(true)) end

-- ==================== Main loop (Aimbot + Speed) ====================
bind(RunService.RenderStepped, function()
    -- Aim
    if (ShootOn or WatchOn) and LPRoot then
        frameCounter = frameCounter + 1
        local refreshLOS = (frameCounter % LOS_REFRESH_FRAMES == 0)

        local head, hasLOS, vel, targetPlayer = findBestTarget(refreshLOS)
        if head then
            local shielded = ShootOn and hasLOS and isShielded(LPRoot.Position, head.Position)
            local canShoot = ShootOn and hasLOS and not shielded
            -- Lock ทันทีที่เจอ LOS (sticky จาก findBestTarget จะคง lock จนเป้าหลบ)
            if canShoot and targetPlayer then
                lockedPlayer = targetPlayer
                lockedHead   = head
                lockedRoot   = pCacheRoot[targetPlayer]
                lockedHum    = pCacheHum[targetPlayer]
            end

            local fromPos = LPRoot.Position
            local aimPos
            if canShoot then
                aimPos = predictHead(head.Position, vel.X, vel.Y, vel.Z, fromPos)
            elseif WatchOn then
                aimPos = getPreAimPosition(fromPos, head.Position, vel.X, vel.Z)
            end

            if aimPos then
                Camera.CFrame = cf_lookAt(Camera.CFrame.Position, aimPos)
                if canShoot then
                    local now = clock()
                    if now - lastShot >= FIRE_RATE then
                        lastShot = now
                        if VIM then
                            pcall(function()
                                VIM:SendMouseButtonEvent(viewportX, viewportY, 0, true,  game, 0)
                                VIM:SendMouseButtonEvent(viewportX, viewportY, 0, false, game, 0)
                            end)
                        end
                    end
                end
            end
        end
    end
    -- (FaceLock ทำงานใน task แยก ด้านล่าง)
end)

-- ==================== Speed loop (Heartbeat) ====================
bind(RunService.Heartbeat, function()
    if not bv or not bv.Parent or not LPHum or LPHum.Health <= 0 then return end
    if SPEED_ON then
        local dir = LPHum.MoveDirection
        if dir.Magnitude > 0.1 then
            bv.MaxForce = v3_new(1e5, 0, 1e5)
            bv.Velocity = dir * SPEED_VAL
        else
            bv.MaxForce = v3_new(0, 0, 0)
            bv.Velocity = v3_new(0, 0, 0)
        end
    else
        bv.MaxForce = v3_new(0, 0, 0)
        bv.Velocity = v3_new(0, 0, 0)
    end
end)

-- ==================== Follow Friend (Anti-Hacker) — P key ====================
-- ตามหลังเพื่อน (teammate) ที่ใกล้สุด ห่าง 10 studs ด้านหลัง
-- ใช้ BodyPosition (ผลักด้วยฟิสิกส์)
local bpGod = nil

-- Animation tracks สำหรับ follow mode (forward-declared ด้านบนแล้ว)
godWalkTrack = nil
godJumpTrack = nil
local GOD_ANIM_RUN  = "rbxassetid://507767714"  -- Roblox default run
local GOD_ANIM_JUMP = "rbxassetid://507765000"  -- Roblox default jump

local function loadGodAnims()
    if not LPChar then return end
    local hum = LPChar:FindFirstChildOfClass("Humanoid")
    if not hum then return end
    local aRun  = Instance.new("Animation"); aRun.AnimationId  = GOD_ANIM_RUN
    local aJump = Instance.new("Animation"); aJump.AnimationId = GOD_ANIM_JUMP
    godWalkTrack = hum:LoadAnimation(aRun)
    godJumpTrack = hum:LoadAnimation(aJump)
    godWalkTrack.Priority  = Enum.AnimationPriority.Action
    godJumpTrack.Priority  = Enum.AnimationPriority.Action
end

function stopGodAnims()
    pcall(function() if godWalkTrack and godWalkTrack.IsPlaying then godWalkTrack:Stop(0.2) end end)
    pcall(function() if godJumpTrack and godJumpTrack.IsPlaying then godJumpTrack:Stop(0.1) end end)
end

local function destroyGodBP()
    if bpGod then pcall(function() bpGod:Destroy() end); bpGod = nil end
    _G.RB06_GOD_BP = nil
    stopGodAnims()
end

bind(RunService.Heartbeat, function()
    if not GOD_ON then
        if bpGod then destroyGodBP() end
        return
    end
    if not LPRoot or not LPHum or LPHum.Health <= 0 then
        if bpGod then destroyGodBP() end
        return
    end

    -- หาเพื่อน (teammate) ที่ใกล้สุด
    local fromPos = LPRoot.Position
    local fpx, fpy, fpz = fromPos.X, fromPos.Y, fromPos.Z
    local myID = getTeamID(LP)
    local nearestRoot, nearestDSq = nil, math_huge
    for i = 1, targetCount do
        local p = activeTargets[i]
        local hum = pCacheHum[p]
        if hum and hum.Health > 0 and myID and getTeamID(p) == myID then
            local root = pCacheRoot[p]
            if root then
                local rp = root.Position
                local dx, dy, dz = rp.X - fpx, rp.Y - fpy, rp.Z - fpz
                local dSq = dx*dx + dy*dy + dz*dz
                if dSq < nearestDSq then nearestDSq, nearestRoot = dSq, root end
            end
        end
    end
    if not nearestRoot then
        if bpGod then destroyGodBP() end
        return
    end

    -- ตำแหน่งหลังเพื่อน 10 studs (ตาม LookVector ของเพื่อน)
    local look = nearestRoot.CFrame.LookVector
    local tp   = nearestRoot.Position
    -- clamp Y: ถ้าเพื่อนต่ำกว่าเรา → ลอยที่ Y เราเอง | ถ้าเพื่อนสูงกว่า → ขึ้นตามไม่จำกัด
    local clampedY  = math_max(tp.Y, LPRoot.Position.Y)
    local targetPos = v3_new(tp.X - look.X * GOD_DIST, clampedY, tp.Z - look.Z * GOD_DIST)

    if not bpGod or not bpGod.Parent then
        if bpGod then pcall(function() bpGod:Destroy() end) end
        bpGod = Instance.new("BodyPosition")
        bpGod.MaxForce = v3_new(4e5, 4e5, 4e5)
        bpGod.P        = 1e4
        bpGod.D        = 1250
        bpGod.Parent   = LPRoot
        _G.RB06_GOD_BP = bpGod
        -- โหลด animation ครั้งแรก
        loadGodAnims()
    end
    bpGod.Position = targetPos

    -- Animation: เดิน / กระโดด / หยุด
    if godWalkTrack and godJumpTrack then
        local myPos   = LPRoot.Position
        local moveDist = math_sqrt(
            (myPos.X - targetPos.X)^2 +
            (myPos.Z - targetPos.Z)^2
        )
        local heightDiff = targetPos.Y - myPos.Y

        if heightDiff > 3 then
            -- ต้องขึ้นสูง → jump animation
            if not godJumpTrack.IsPlaying then
                if godWalkTrack.IsPlaying then godWalkTrack:Stop(0.1) end
                godJumpTrack:Play(0.1)
            end
        elseif moveDist > 1.5 then
            -- กำลังเดินตาม → walk/run animation
            if not godWalkTrack.IsPlaying then
                if godJumpTrack.IsPlaying then godJumpTrack:Stop(0.1) end
                godWalkTrack:Play(0.2)
            end
        else
            -- ถึงที่แล้ว → หยุด
            stopGodAnims()
        end
    end
end)

-- ==================== BackFace (task.defer — รันหลัง script ทุกตัวในเฟรม) ====================
-- task.defer = ต่อท้าย queue ของเฟรมนั้น → เกมเซ็ต CFrame ก่อน → เราเขียนทับทีหลังเสมอ
local function doFaceLock()
    if not FACELOCK_ON or not LPRoot or not LPHum or LPHum.Health <= 0 then return end
    local flHead = findBestTarget(false)
    if not flHead then return end
    local tp    = flHead.Position
    local myPos = LPRoot.Position
    local dx, dz = tp.X - myPos.X, tp.Z - myPos.Z
    if dx*dx + dz*dz > 0.25 then
        -- หน้าชี้ออกจากศัตรู = หลังหันเข้าหาศัตรู
        LPRoot.CFrame = cf_lookAt(myPos, v3_new(myPos.X - dx, myPos.Y, myPos.Z - dz))
    end
end

bind(RunService.Heartbeat, function()
    if FACELOCK_ON then task.defer(doFaceLock) end
end)
